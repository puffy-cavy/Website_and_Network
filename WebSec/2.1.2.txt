USE project2;
CREATE TABLE users (id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, username VARCHAR(32) NOT NULL UNIQUE, password VARCHAR(32) NOT NULL, passwordhash BLOB(16) NOT NULL);
CREATE TABLE history (id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, user_id INT UNSIGNED NOT NULL, query VARCHAR(2048) NOT NULL);
CREATE UNIQUE INDEX index_name ON users (username);
CREATE INDEX index_name ON history (user_id);
